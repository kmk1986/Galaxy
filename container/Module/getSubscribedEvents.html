Module.getSubscribedEvents

returns an array of subscribed events

Usage

Module.getSubscribedEvents()

Example


var Module = (function(){

    var mediator = Galaxy.register('testModule');

    mediator.subscribe({
    	'onReady': [onReady, this]
	});


	// outputs ['onReady']
	console.log( mediator.getSubscribedEvents() );	
	

}());
