Galaxy.register

Creates a facade object and return it

Usage

Galaxy.register('your id')

Parameter Type Description
id String id of your facade instance

Example

    var GUIWindow = (function(){

        // notifier gets all the methods from Module (facade)
        var notifier = Galaxy.register('test')

    }());

Galaxy.registerSingleton

Attempts to create a facade object with an id. If the Mediator already has a facde object with the id provided, it simply returns it.

Example

    
    var CartItem = function() {

        // all the CartItem instances will have the same facade instance
        this.notifier = Galaxy.registerSingleton('cartItem');

    }

    var Cart = (function(){

        var notifier = Galaxy.register('cart');
        var items = [];

        var addItem = fuction( eventData ) {

            items.push( new CartItem() );

        }

        notifier.subscribe({
            'addItem': [addItem, this]
        });

    }());

Galaxy.on

Bind to an internal event. on/off are used to make a plugin

Usage

Galaxy.on(eventName, callback)

Parameter Type Description
eventName String name of the internal event you're interested in
callback Function a callback for the event

Example

The following code snippet will console.log out whenever an event has been subscribed/published. 


;(function(Galaxy){
    
    /*
    * @param {object} eventData
    *   @param {object} sender
    *       @param {string} namespace - sender object
    *       @param {string} method - sender method
    *   @param {string} eventName - name of the event published
    *   @param {string} senderName - name of the sender who published  the event
    *   @param {string} moduleName - name of the module who received the event
    *   @param {object} eventData 
    */
    
    Galaxy.on('event-published', function(eventData){

        var msg = [
            eventData.senderName,
            " published '",
            eventData.eventName,
            "' to ",
            eventData.moduleName,
            " with eventData of ",
            eventData.eventData
        ];

        eventData = null;

        console.log(msg.join(""));

    });


    /*
    * @param {object} eventData
    *   @param {object} sender
    *       @param {string} namespace - sender object
    *       @param {string} method - sender method
    *   @param {string} eventName - name of the event subscribed
    *   @param {string} moduleName - name of the module who subscribed the event
    */  
    Galaxy.on('event-subscribed', function(eventData){

        var msg = [

            eventData.moduleName,
            " subscribed to ",
            eventData.eventName

        ];

        eventData = null;

        console.log( msg.join("") );
        
    });



}(Galaxy));

Galaxy.off

Unbind an event from the internal events

Usage

Galaxy.off(eventName, callback)

Parameter Type Description
eventName String name of the internal event you've subscribed
callback Function The callback you passed to Galaxy.on method

Example


;(function(Galaxy){
    
    /*
    * @param {object} eventData
    *   @param {object} sender
    *       @param {string} namespace - sender object
    *       @param {string} method - sender method
    *   @param {string} eventName - name of the event published
    *   @param {string} senderName - name of the sender who published  the event
    *   @param {string} moduleName - name of the module who received the event
    *   @param {object} eventData 
    */

    var onEventPublished = function(eventData){

        var msg = [
            eventData.senderName,
            " published '",
            eventData.eventName,
            "' to ",
            eventData.moduleName,
            " with eventData of ",
            eventData.eventData
        ];

        eventData = null;

        console.log(msg.join(""));



    }

    Galaxy.on('event-published', onEventPublished );

    // pseudo code
    // when a user clicks 'stop logging', stop it.

    $("#stoppLogging").on('click',function(){

        Galaxy.off("event-published", onEventPublished );

    });



}(Galaxy));

Galaxy.getListenersByEventName

returns a lsit of module ids who has subscribed to $eventName

Usage

Galaxy.getListenersByEventName(eventName)

Parameter Type Description
eventName String name of the event

Returns an array of module ids

Example

 
var CartItem = function(options) {

}

var Cart = (function(){

    var mediator = Galaxy.register('Cart');

    var items = [];

    mediator.subscribe({
        'addToCart': [this.addToCart]
    });

    return {

        addToCart: function(eventData) {
            items.push( new CartItem(eventData) );
        }
    }


}());

var PageController = (function(){

    var mediator = Galaxy.register('Page');

    mediator.reserveEvent("addToCart");

    var onAddToCartClicked = function() {

        mediator.publish("addToCart",{});
    
    }

    $("#addToCart").on("click", onAddToCartClicked );


}());

// Galaxy.getListenersByEventName("addToCart") outputs
// ["Cart"]

Mediator.start

starts a module. If the module is in 'pause' status, it will resume. If the module has never been started, it will initiailize the module.

Usage

Module.start()

Example


var Module = (function(){
    
    // you must pass falto as the secnd parameter of the register method
    // the default is 'true' so that it will autostart all the time
    var mediator = Galaxy.register('testModule' , false);

    // this will fail since the mediator is not started yet.
    mediator.subscribe({
    	'onReady': [this.onReady, this]
	});

	mediator.start();
	

}());

Mediator.stop

stops a module. All the module will be removed from the mediator. Once you call stop, you will have to re-initialize it with .register to re-use it. You also have to call .subscribe again. Basically, calling stop() makes a module object unusable.

Usage

Module.stop()

Example


var Module = (function(){

    var mediator = Galaxy.register('testModule');

    var onReady = function() {

    	// stop this module when we get 'onReady' event
    	mediator.stop();
	}


    mediator.subscribe({
    	'onReady': [onReady, this]
	});


}());

var Cart = (function(){

    var onAddToCart = function(eventData) {

    	this.mediator.unsubscribe("addToCart");

	}


    mediator.subscribeOne("addToCart", onAddToCart, this);
   

}());

Module.subscribe

subscribe to events

Usage

Module.subscribe(object);

Module.subscrbe({
	"event name goes here": this.callback,
	"event name goes here": [this.callback, context]
})

Dev Notes

Module.subscribe is a facade method for Module.subscribeOne. It invokes subscribeOne to subscribe an event.

Parameter

subscribe expects an object (collection of event and callback paris).

Parameter Type Description
eventName String name of the event
callback a function or an array with a function and a context depends on what you want you can pass either a function or a function with a context in an array. Refer to the example or usage.

Example


var Cart = (function(){
    
    var mediator = Galaxy.register('testModule');
    this.name = 'Cart';

    var onAddToCart = function(eventData) {

    	//when you use 'this' keyword here, it refers to the window object
    	this.mediator.unsubscribe("addToCart"); // fails

	}

	var onRemoveFromCart = function(eventData) {

		console.log( this.name ); // outputs Cart as expected

	}

    mediator.subscribe({
    	'addToCart': onAddToCart,
    	'removeFromCart': [onRemoveFromCart, this]
	});
   

}());

Module.publish

publishes an event

Usage

Module.publish(eventName, eventData);

Note

Parameter

Parameter Type Description
eventName String name of the event
eventData Mixed a data for the event

Example


var Cart = (function(){
    
    var mediator = Galaxy.register('testModule');
    this.name = 'Cart';

    var onAddToCart = function(eventData) {

        // code goes here
        // let other modules know that Cart has added an item
        // note that we've already reserved 'ItemAdded'

        mediator.publish("ItemAdded",{
            itemName: "test item",
            price: 33
        });


	}

	var onRemoveFromCart = function(eventData) {

		console.log( this.name ); // outputs Cart as expected

	}

    mediator.reserveEvents(["ItemAdded"]);

    mediator.subscribe({
    	'addToCart': onAddToCart,
    	'removeFromCart': [onRemoveFromCart, this]
	});
   

}());
